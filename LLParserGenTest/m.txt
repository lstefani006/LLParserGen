

fun ww(a,b) {
	var d;
	a = ww(a, b+1);
	ww(a, b+1);
	fact(3);
	d = 3;
	if (d > 0)
		a = 3;
	
	while (Fibonacci(a) > a)
		a = a-1;
}

fun fact(n) {
	if (n == 0) return 1;
	return n * fact(n-1);
}

fun Fibonacci(n)
{
	if (n == 0)
		return 0;
	else if (n == 1)
		return 1;
	
	return Fibonacci(n-1) + Fibonacci(n-2);
	n = 0;
}
